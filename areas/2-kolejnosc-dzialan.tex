% -- 11 --
\ExerciseSlide[\CategoryBadge[OpsColor!20]{Podstawy}]
  {2}{60 s}{$3 + 4 \times 2 = ?$}

\AnswerSlide[\CategoryBadge[OpsColor!20]{Podstawy}]{11}{
  {\footnotesize
    \[ 3 + \underbrace{4 \times 2}_{8} = 3 + 8 = \mathbf{11} \]
    Mnożenie przed dodawaniem (BODMAS / PEMDAS).
  }
}

% -- 12 --
\ExerciseSlide[\CategoryBadge[OpsColor!20]{Nawiasy}]
  {2}{60 s}{$(3 + 4) \times 2 = ?$}

\AnswerSlide[\CategoryBadge[OpsColor!20]{Nawiasy}]{14}{
  {\footnotesize
    \[ \underbrace{(3 + 4)}_{7} \times 2 = 7 \times 2 = \mathbf{14} \]
    Nawiasy obliczamy jako pierwsze!
  }
}

% -- 13 --
\ExerciseSlide[\CategoryBadge[OpsColor!20]{Mieszane}]
  {2}{90 s}{$20 \div (2 + 3) \times 4 = ?$}

\AnswerSlide[\CategoryBadge[OpsColor!20]{Mieszane}]{16}{
  {\footnotesize
    \begin{align*}
      20 \div \underbrace{(2{+}3)}_{5} \times 4
        &= 20 \div 5 \times 4 \\
        &= 4 \times 4 = \mathbf{16}
    \end{align*}
    Działania tego samego priorytetu --- od lewej do prawej.
  }
}

% -- 14 --
\ExerciseSlide[\CategoryBadge[OpsColor!20]{Mieszane}]
  {2}{90 s}{$15 - 3 \times 4 + 8 \div 2 = ?$}

\AnswerSlide[\CategoryBadge[OpsColor!20]{Mieszane}]{7}{
  {\footnotesize
    \begin{align*}
      15 - \underbrace{3 \times 4}_{12} + \underbrace{8 \div 2}_{4}
        &= 15 - 12 + 4 = \mathbf{7}
    \end{align*}
  }
}

% -- 15 --
\ExerciseSlide[\CategoryBadge[OpsColor!20]{Nawiasy}]
  {3}{2 min}{$(8+12)\div 4 - 3\times(7-5) = ?$}

\AnswerSlide[\CategoryBadge[OpsColor!20]{Nawiasy}]{$-1$}{
  {\footnotesize
    \begin{align*}
      \underbrace{(8{+}12)}_{20} \div 4
        - 3 \times \underbrace{(7{-}5)}_{2}
        &= 20 \div 4 - 3 \times 2 \\
        &= 5 - 6 = \mathbf{-1}
    \end{align*}
  }
}

% -- 16 --
\ExerciseSlide[\CategoryBadge[OpsColor!20]{Potęgi}]
  {3}{2 min}{$2^3 + 6 \times (4-2) \div 3 = ?$}

\AnswerSlide[\CategoryBadge[OpsColor!20]{Potęgi}]{12}{
  {\footnotesize
    \begin{align*}
      \underbrace{2^3}_{8} + 6 \times \underbrace{(4{-}2)}_{2} \div 3
        &= 8 + \underbrace{6 \times 2}_{12} \div 3 \\
        &= 8 + 4 = \mathbf{12}
    \end{align*}
  }
}

% -- 17 --
\ExerciseSlide[\CategoryBadge[OpsColor!20]{Nawiasy}]
  {3}{2 min}{$100 - (5\times 8 + 4\times 3) = ?$}

\AnswerSlide[\CategoryBadge[OpsColor!20]{Nawiasy}]{48}{
  {\footnotesize
    \begin{align*}
      100 - (\underbrace{5{\times}8}_{40} + \underbrace{4{\times}3}_{12})
        &= 100 - 52 = \mathbf{48}
    \end{align*}
  }
}

% -- 18 --
\ExerciseSlide[\CategoryBadge[OpsColor!20]{Potęgi}]
  {3}{3 min}{$3\times(2+4)^2 \div 12 - 1 = ?$}

\AnswerSlide[\CategoryBadge[OpsColor!20]{Potęgi}]{8}{
  {\footnotesize
    \begin{align*}
      3 \times \underbrace{(2{+}4)^2}_{36} \div 12 - 1
        &= \underbrace{3 \times 36}_{108} \div 12 - 1 \\
        &= 9 - 1 = \mathbf{8}
    \end{align*}
  }
}
