\ExerciseSlide[\CategoryBadge[MixColor!20]{Pr\k{e}dko\'s\'c}]
  {2}{2 min}{%
    Poci\k{a}g jedzie 120~km w 1{,}5~godziny.\par
    Jaka jest jego \'{s}rednia pr\k{e}dko\'s\'c?
  }

\AnswerSlide[\CategoryBadge[MixColor!20]{Pr\k{e}dko\'s\'c}]{80~km/h}{
  {\footnotesize
    \[ v = \frac{d}{t} = \frac{120\ \text{km}}{1{,}5\ \text{h}}
         = \mathbf{80\ \text{km/h}} \]
    Zamiana: $1{,}5\ \text{h} = 90\ \text{min}$;
    $120 \div 1{,}5 = 80$.
  }
}

% -- 35 --
\ExerciseSlide[\CategoryBadge[MixColor!20]{Procenty}]
  {2}{2 min}{%
    Pizza ma 8 kawa\l{}k\'{o}w, zjadasz 3.\par
    Jaki procent pizzy zostaje?
  }

\AnswerSlide[\CategoryBadge[MixColor!20]{Procenty}]{62{,}5\%}{
  {\footnotesize
    Zostaje $8 - 3 = 5$ kawa\l{}k\'{o}w.
    \[ \frac{5}{8} \times 100\% = \mathbf{62{,}5\%} \]
  }
}

\ExerciseSlide[\CategoryBadge[MixColor!20]{Geometria}]
  {3}{3 min}{%
    Pok\'{o}j ma wymiary $5\,\text{m}\times 4\,\text{m}$.\par
    Uk\l{}adasz kafelki $50\,\text{cm}\times 50\,\text{cm}$.\par
    Ile kafelk\'{o}w potrzebujesz?
  }

\AnswerSlide[\CategoryBadge[MixColor!20]{Geometria}]{80 kafelk\'{o}w}{
  {\footnotesize
    \begin{itemize}
      \item Wzd\l{}u\. z: $5\,\text{m} \div 0{,}5\,\text{m} = 10$ kafelk\'{o}w
      \item W poprzek: $4\,\text{m} \div 0{,}5\,\text{m} = 8$ kafelk\'{o}w
    \end{itemize}
    \[ 10 \times 8 = \mathbf{80\ \text{kafelk\'{o}w}} \]
  }
}

\ExerciseSlide[\CategoryBadge[MixColor!20]{Finanse}]
  {3}{3 min}{%
    Inwestujesz 1000~z\l{} przy 5\% odsetek rocznie.\par
    Ile masz po 2~latach\par
    (procent sk\l{}adany)?
  }

\AnswerSlide[\CategoryBadge[MixColor!20]{Finanse}]{1102{,}50~z\l{}}{
  {\footnotesize
    Wz\'{o}r na procent sk\l{}adany: $K = K_0 \cdot (1+r)^n$
    \begin{align*}
      K &= 1000 \times (1{,}05)^2 \\
        &= 1000 \times 1{,}1025 \\
        &= \mathbf{1102{,}50\ \text{z\l{}}}
    \end{align*}
    Po roku: $1050$~z\l{}, po dwu: $1050 \times 1{,}05 = 1102{,}50$~z\l{}.
  }
}

\ExerciseSlide[\CategoryBadge[MixColor!20]{Praca}]
  {3}{5 min}{%
    3 robotnik\'{o}w wykonuje prac\k{e} w 6 dni.\par
    Ile dni zajmie ta praca\par
    2 robotnikom?
  }

\AnswerSlide[\CategoryBadge[MixColor!20]{Praca}]{9 dni}{
  {\footnotesize
    \L{}\k{a}czna praca: $3 \times 6 = 18$ robotnikodni.
    \[ \text{Czas} = \frac{18\ \text{robotnikodni}}{2\ \text{robotnik\'{o}w}}
         = \mathbf{9\ \text{dni}} \]
    Wi\k{e}cej robotnik\'{o}w = kr\'{o}cej; mniej = d\l{}u\. zej.
  }
}
